  </xsd:sequence>
  <xsd:attribute name="lineno" type="xsd:integer"/>
  <xsd:attribute name="refid" type="xsd:string"/>
  <xsd:attribute name="refkind" type="DoxRefKind"/>
  <xsd:attribute name="external" type="DoxBool"/>
</xsd:complexType>
