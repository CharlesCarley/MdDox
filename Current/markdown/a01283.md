<br/>

<span class="bold-text"><b>docBlockQuoteType</b></span>
<span class="inline-text"> scaffolding. </span>
<br/>
<br/>

```xml
<xsd:complexType name="docBlockQuoteType">
  <xsd:sequence>
    <xsd:element minOccurs="0" name="para" type="docParaType" maxOccurs="unbounded"/>
  </xsd:sequence>
</xsd:complexType>
```
