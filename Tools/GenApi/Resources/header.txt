#pragma once
#include "Doxygen.h"
${includes}

namespace MdDox::Doxygen
{
${visitor}
    /**
     * \brief Implements the <tt>${tagName}</tt> scaffolding.
     *
     * The following xml provides the source for the <b>${tagName}</b> 
     * scaffolding.
     *
     * \code{.xml}${xmlSource}
     * \endcode
     */
    class ${className}
    {
    private:
        Xml::Node* _node;

    public:
        ${className}() :
            _node(nullptr)
        {
        }

        explicit ${className}(Xml::Node* node) :
            _node(node)
        {
        }

        ${className}(const ${className}& other) = default;

        /**
         * \brief Checks to see if the internal node is usable.
         */
        bool isValid() const;

        /**
         * \brief Checks to see if the internal node has children.
         */
        bool empty() const;

        /**
         * \brief Resets the internal node pointer to null.
         */
        void reset();

        /**
         * \brief Returns the wrapped node pointer.
         */
        Xml::Node* node() const;

        /**
         * \brief Sorts the nodes in the query based on priority of type.
         */
        void sort(const SortMap& childOrder) const;

        /**
         * \brief Provides access to the node's inner text.
         * \return The text data associated with this node the default value.
         */
        const String& text(const String& def="") const;
${swapCode}
    };

    inline bool ${className}::isValid() const
    {
        return _node != nullptr;
    }

    inline void ${className}::reset()
    {
        _node = nullptr;
    }

    inline Xml::Node* ${className}::node() const
    {
        return _node;
    }

    inline bool ${className}::empty() const
    {
        return _node? !_node->hasChildren() : true;
    }

} // namespace MdDox::Doxygen
